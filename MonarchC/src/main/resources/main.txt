function main(): void {
    var age: byte;
    while (true) {
        var c: byte;
        c = scan;
        if (c == '\n') {
            break;
        }
        age = age * 10b;
        age = age + c - '0';
    }

    if (age > 17b) {
        print "OLD ENOUGH TO VOTE\n";
    } else {
        print "WAIT ";
        var yearsToWait: byte;
        yearsToWait = 18b - age;
        while (yearsToWait > 0b) {
            print yearsToWait % 10b + '0';
            yearsToWait = yearsToWait / 10b;
        }
        print " YEARS TO VOTE\n";
    }
}
