function main(): void {
    var word = &! 30b 0b as &byte;
    var wordLen: byte;

    print "Enter a word: ";
    scanString(word, &wordLen);
    print "\n";

    var line = 0b;
    while (line < wordLen) {
        printString(word, wordLen); print "\n";
        line = line + 1b;
    }

    *! word;
}

function scanString(buffer: &byte, len: &byte): void {
    var cIndex = 0b;

    while (cIndex < 30b) {
        var c = scan;
        if (c == '\n') {
            break;
        }
        var cPtr = buffer + cIndex;
        *cPtr = c;
        cIndex = cIndex + 1b;
    }

    *len = cIndex;
}

function printString(str: &byte, len: byte): void {
    var cIndex = 0b;
    while (cIndex < len) {
        var cPtr = str + cIndex;
        print *cPtr;
        cIndex = cIndex + 1b;
    }
}
